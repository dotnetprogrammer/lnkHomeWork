아래와 같이 시작하는 코드, 끝나는 코드를 정해놓고 transition속성이 애니메이션을 생성해 줌.


#motionBox {
	-webkit-transition: margin-left 1s ease-in-out, 
	opacity 1s ease-in-out;
}
#motionBox.left {margin-left:0%;}
#motionBox.right {margin-left:90%;}

객체의 지속적인 움직임 및 변화


$(function(){
	function growUp(){
		$(".css2 .move_2_obj .motionBox").animate({
			height:"150px",
			width:"150px",
			opacity:1
		}, 1000, function(){
			growDn();
		});
	}
	function growDn(){
		$(".css2 .move_2_obj .motionBox").animate({
			height:"50px",
			width:"50px",
			opacity:0.2
		}, 1000, function(){
			growUp();
		});
	}
	growUp()
});

객체의 지속적인 움직임.


@-webkit-keyframes pulse {
	from {
		opacity: 0.2;
		width: 50px;
		height: 50px;
	}
	to {
		opacity: 1;
		width: 150px;
		height: 150px;
	}
}
.css3 .move_2_obj .motionBox {
	-webkit-animation-name: pulse;
	-webkit-animation-duration: 1s;
	-webkit-animation-iteration-count: infinite;
	-webkit-animation-timing-function: ease-in-out;
	-webkit-animation-direction: alternate;
}

@-webkit-keyframes rotated { 
	0% { -webkit-transform: rotate(0deg); }
	100% { -webkit-transform: rotate(360deg); }
